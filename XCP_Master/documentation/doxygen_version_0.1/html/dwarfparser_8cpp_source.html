47 void DwarfParser::resolveVariables(const dwarf::die& node, bool isParentNodeSubprogram) {
48  uint64_t sectionOffset = node.get_section_offset();
49 
50  if(node.tag == dwarf::DW_TAG::subprogram) {
51  isParentNodeSubprogram = true;
52  } else if(node.tag == dwarf::DW_TAG::compile_unit) {
53  isParentNodeSubprogram = false;
54  }
55 
56 
57  if(!isParentNodeSubprogram) { //ignore local variables from methods/functions
58  if(node.tag == dwarf::DW_TAG::compile_unit) {
59  lookupSrcPathCompileUnit(node);
60  } else if(node.tag == dwarf::DW_TAG::structure_type) {
61  lookupStructs(node);
62  } else if((node.tag == dwarf::DW_TAG::variable) || (node.tag == dwarf::DW_TAG::member)) {
63  if(isVariableLocationValid(node) || (node.tag == dwarf::DW_TAG::member)) { //@TODO refactor code (gcc 4.6 and gcc 4.9 compliance)
64  shared_ptr<DwarfTreeObj> dwarfTreeObj = std::make_shared<DwarfTreeObj>();

66  addInfoToDwarfTreeObj(node, *dwarfTreeObj);
67  addDwarfTreeObjToMap(sectionOffset, dwarfTreeObj);
68  }
69  }
70  }
71 
72  for (const auto& child : node) {
73  resolveVariables(child, isParentNodeSubprogram);
74  }

137 void DwarfParser::lookupStructElementsInDwarfTree(DwarfTreeObj &dwarf, uint64_t dwarfKey, int depth) {
138 
139  if(isStructDepthLevelZero(dwarf) || depth > 0) {
140  string parentDataType = dwarf.getDataType().first;
141  string srcFileParent = dwarf.getSourcePath();
142 
143  if(parentDataType.size() > 0) {
144  map<uint64_t, shared_ptr<DwarfTreeObj>>::reverse_iterator itRevDwarfTreeElement(dwarfTree.find(dwarfKey));
145  vector<uint64_t> dwarfStructKeys;
146  for (auto &it = itRevDwarfTreeElement; it != dwarfTree.rend(); ++it) {
147  shared_ptr<DwarfTreeObj> dwarfChild = it->second;
148 
149  if((isChildOfParentStruct(parentDataType, dwarfChild->getStructName())) && isStructChildValid(*dwarfChild)) {
150  uint64_t key = it->first;
151  dwarfStructKeys.push_back(key);
152 
153  if(dwarfChild->getAbsType() == AbstractType::Struct) {
154  if(depth <= STRUCT_MEMBER_SEARCH_DEPTH_MAX) {
155  lookupStructElementsInDwarfTree(*dwarfChild, key, ++depth);
156  }
157  }
158  }
159 
160  if(isOutOfScope(srcFileParent, dwarfChild->getSourcePath())) {
161  break;
162  }
163  }
164  addStructElementsToStruct(dwarf, dwarfStructKeys);
165  }
166  }

183 void DwarfParser::addInfoToDwarfTreeObj(const dwarf::die& node, DwarfTreeObj& dwarfTreeObj) {
184  for (const auto &attr : node.attributes()) {
185  dwarf::DW_AT attributeName = attr.first;
186  string value = to_string(attr.second);
187 
188  if(attributeName == dwarf::DW_AT::name) {
189  dwarfTreeObj.setAttributeName(value);
190  }
191  if(attributeName == dwarf::DW_AT::decl_file) {
192  dwarfTreeObj.setSourcePath(srcPathCompileUnitLast);
193  }
194  if(attributeName == dwarf::DW_AT::decl_line) {
195  dwarfTreeObj.setDeclLine(stringHexToDec(value));
196  }
197  if(attributeName == dwarf::DW_AT::type) {
198  dwarfTreeObj.setDataType(lookupDataType(node));
199  }
200  }
201 
202  if(node.tag == dwarf::DW_TAG::member) {
203  AbstractType absType = lookupAbstractTypeForVariable(node);
204  dwarfTreeObj.setAbsType(absType);
205  dwarfTreeObj.setStructName(structNameLast);
206  } else if(node.tag == dwarf::DW_TAG::variable) {
207  AbstractType absType = lookupAbstractTypeForVariable(node);
208  dwarfTreeObj.setAbsType(absType);
209  dwarfTreeObj.setAddr(lookupAddrOfVariable(node, absType));
210  }

214 std::pair<string, uint32_t> DwarfParser::lookupDataType(const dwarf::die& node) {
215  string name;
216  uint32_t byteSize = 0;
217  std::pair <string, uint32_t> ret;
218 
219  dwarf::value value = node.resolve(dwarf::DW_AT::type);
220  if(value.valid()) {
221  dwarf::die nodeChild = value.as_reference();
222 
223  if((nodeChild.tag == dwarf::DW_TAG::base_type) || (nodeChild.tag == dwarf::DW_TAG::structure_type)) {
224  for(auto &attr : nodeChild.attributes()) {
225  dwarf::DW_AT attributeName = attr.first;
226  string value = to_string(attr.second);
227  if(attributeName == dwarf::DW_AT::name) {
228  name = value;
229  }
230  if(attributeName == dwarf::DW_AT::byte_size) {
231  byteSize = stringHexToDec(value);
232  }
233  }
234  ret = std::make_pair(name, byteSize);
235  } else {
236  ret = lookupDataType(nodeChild);
237  }
238  }
239  return ret;

242 AbstractType DwarfParser::lookupAbstractTypeForVariable(const dwarf::die& node) {
243  AbstractType retType = AbstractType::NotFound;
244 
245  dwarf::value value = node.resolve(dwarf::DW_AT::type);
246  if(value.valid()) {
247  dwarf::die nodeChild = value.as_reference();
248 
249  if(nodeChild.tag == dwarf::DW_TAG::array_type) {
250  retType = AbstractType::Array;
251  } else if(nodeChild.tag == dwarf::DW_TAG::structure_type) {
252  retType = AbstractType::Struct;
253  } else if(nodeChild.tag == dwarf::DW_TAG::pointer_type) {
254  retType = AbstractType::Pointer;
255  } else if(nodeChild.tag == dwarf::DW_TAG::base_type) {
256  retType = AbstractType::Variable;
257  } else {
258  retType = lookupAbstractTypeForVariable(nodeChild);
259  }
260  }
261  return retType;

264 uint64_t DwarfParser::lookupAddrOfVariable(const dwarf::die &node, AbstractType absType) {
265  uint64_t addr = 0;
266  if ((node.tag == dwarf::DW_TAG::variable) && (absType != AbstractType::Pointer)) { //dwarf library does not support pointers yet
267  dwarf::value location = node.resolve(dwarf::DW_AT::location);
268 
269  if (isVariableLocationValid(node)) {
270  try {
271  dwarf::expr_result result = location.as_exprloc().evaluate(&dwarf::no_expr_context);
272  addr = result.value;
273  } catch(dwarf::expr_error& caught) {
274  std::cout << "Error lookupAddrOfVariable(): " << caught.what() << std::endl;
275  }
276  }
277  }
278  return addr;
279 }
